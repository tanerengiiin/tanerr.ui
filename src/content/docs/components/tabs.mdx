## Usage

```tsx
import {
  Tabs,
  TabsList,
  TabsPanel,
  TabsTab
} from "@/components/ui/tabs";
```

```tsx
<Tabs defaultValue="tab1">
  <TabsList>
    <TabsTab value="tab1">Tab 1</TabsTab>
    <TabsTab value="tab2">Tab 2</TabsTab>
  </TabsList>
  <TabsPanel value="tab1">
    Content for Tab 1
  </TabsPanel>
  <TabsPanel value="tab2">
    Content for Tab 2
  </TabsPanel>
</Tabs>
```
