## Usage

```tsx
import {
  Tooltip,
  TooltipPopup,
  TooltipPositioner,
  TooltipProvider,
  TooltipTrigger,
} from "@/components/ui/tooltip";
```

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger
      render={
        <Button variant={"ghost"} size={"icon"}>
          <Bold />
        </Button>
      }
    />
    <TooltipPositioner>
      <TooltipPopup>
        <p>Bold</p>
      </TooltipPopup>
    </TooltipPositioner>
  </Tooltip>
</TooltipProvider>
```
