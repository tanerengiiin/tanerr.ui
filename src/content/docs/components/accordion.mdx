## Usage

```tsx 
import {
    Accordion,
    AccordionItem,
    AccordionPanel,
    AccordionTrigger,
} from "@/components/ui/accordion";
```
```tsx 
<Accordion>
    <AccordionItem>
        <AccordionTrigger>
        What payment methods do you accept?
        </AccordionTrigger>
        <AccordionPanel>
        We accept all major credit and debit cards.
        </AccordionPanel>
    </AccordionItem>
</Accordion>
```
